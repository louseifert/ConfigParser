ConfigParser is a handfull of classes designed to make parsing configuration files and commandline arguements straight forward presenting a consistent interface with a few security controls to the programmer.

	int main(int c, char **arg) {
  		ConfigParser *config = new ConfigParser(c, arg);
		string variable=config->get_string("cmdSetting");
		std::cout << "the command line option -cmdSetting= was set to:" << variable << std::endl;
	        config->load_ini("path/to/ini");
		delete(config);
		return 0;
	}
