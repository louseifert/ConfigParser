27std::vector<std::string> splitstring(std::string *p, std::string *d) {
28 std::vector<std::string> rvec;
29 size_t last = 0;
30 for (size_t i = 0; i < p->size(); i++) {
31 if (p->at(i) != d->at(0)) {
32 continue;
33 } else {
34 if (d->size() == 1) {
35 rvec.push_back(p->substr(last, i - last));
36 last = i + d->size();
37 i = last;
38
39 continue;
40 } else {
41 for (size_t j = 0; j < d->size(); j++) {
42 if (j + i >= p->size()) {
43 rvec.push_back(p->substr(last, p->size()));
44 }
45 if (p->at(i + j) != d->at(j)) {
46 continue;
47 } else {
48 rvec.push_back(p->substr(last, i - last));
49 last = i + d->size();
50 i = i + d->size();
51 continue;
52 }
53 }
54 }
55 }
56 }
57 if (last != p->size()) {
58 rvec.push_back(p->substr(last, p->size() - last));
59 }
60 return rvec;
61}
68std::string str_between(std::string *p, std::string *start, std::string *end) {
69 size_t s = p->find(*start);
70 size_t e = p->find(*end);
71 return p->substr(s + start->size(), e - s - start->size());
72}
