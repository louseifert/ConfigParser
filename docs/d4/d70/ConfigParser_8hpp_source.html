188 void load_ini(string filename) {
189 if (ini_loaded) {
190 throw security_exception("you can only load an ini file once");
191 }
196 if (!std::filesystem::exists(filename) ||
197 !std::filesystem::is_regular_file(filename)) {
198 throw file_access_exception(
199 (const string) "file either does not exist or is not a regular file");
200 }
201
202 // Throw a security exception if the file is writable
203 auto perm = std::filesystem::status(filename).permissions();
204 if ((perm == std::filesystem::perms::others_write ||
205 perm == std::filesystem::perms::group_write ||
206 perm == std::filesystem::perms::owner_write ||
207 perm == std::filesystem::perms::group_exec ||
208 perm == std::filesystem::perms::owner_exec ||
209 perm == std::filesystem::perms::others_exec ||
210 perm == std::filesystem::perms::others_read) &&
211 !inisec_override) {
212 throw security_exception(
213 (const string) "Setting config files to read only is more secure, or "
214 "use override constructor");
215 }
216 std::string line;
217
221 std::ifstream file(filename, std::ios::in);
222 /*not sure this condition could happen?
223 * if (!file.is_open()) {
224 throw file_access_exception(
225 "File System Error: cannot open, potentially permission");
226 }*/
227
228 string section_head = "";
229 while (std::getline(file, line)) {
230 if (line.empty()) {
231 continue;
232 }
233 while (line[0] == ' ' || line[0] == '\t') {
234 line.erase(0, 1);
235 }
236 if (line.size() == 0 || line[0] == '#') {
237 continue;
238 }
239 if (line[0] == ';' ||
240 (line.length() > 2 && line[0] == '/' && line[1] == '/')) {
241 continue;
242 }
243 /*TODO add regex check here*/
244 if (line[0] == '[') {
245 section_head = line.substr(1, line.find_first_of(']') - 1) + ".";
246 continue;
247 }
248
249 const size_t s = line.find_first_of("=", 0);
250 if (s == line.npos) {
251 flags.insert(line);
252 continue;
253 }
254
255 string key = line.substr(0, s);
256 string value = line.substr(s + 1, line.length());
257 string_ops::strip_qoutes(&key);
258 string_ops::strip_qoutes(&value);
259 key = section_head + key;
260
261 if (!stringops.strip_check(&key) || !stringops.strip_check(&value)) {
262 restricted.emplace(key, value);
263 restricted_count++;
264 } else {
265 if (options.contains(key)) {
266 if (allow_multi) {
267 options_count[key] = options_count[key] + 1;
268 options.emplace(key, value);
269 } else {
270 throw security_exception(
271 "Duplicate key is being added and allow multi is not set: " +
272 key);
273 }
274 } else {
275 options.emplace(key, value);
276 ini_parm_count++;
277 }
278 }
279 }
280 ini_loaded = true;
281 }

320 void parse(int c, char *v[]) {
321
322 string prev = "";
323 string_ops sops;
324 int x=0;
325 while ( x < c ) {
326 //std::cout << "-------------------->x=" << x << " c=" << c << "-------------------------" << std::endl;
327 string str(sops.trim(v[x]));
328 string key;
329 string value;
330 size_t flag = str.find_first_of('-', 0);
331 size_t assigner = str.find_first_of('=', 0);
332 size_t space = str.find_first_of(' ', 1);
333 if ((flag == 0 && (space != str.npos || assigner != str.npos)) ||
334 (flag == 0 && x + 1 < c && v[x + 1][0] != '-')) {
335 int t = x;
336 //std::cout << "f-"<<flag<<" a-"<<assigner << " s-" << space << std::endl;
337 // if there is an equal sign in the string split it and copy the values
338 if (assigner != str.npos) {
339 key = str.substr(1, assigner-1);
340 x++;
341 std::string val = str.substr(assigner + 1, str.size());
342 string_ops::trim(&val);
343 string_ops::strip_qoutes(&val);
344 value.append(val);
345
346 }else if (space != str.npos) {
347 key = str.substr(1, space-1);
348 x++;
349 std::string val = str.substr(space, str.size());
350 string_ops::trim(&val);
351 string_ops::strip_qoutes(&val);
352 value.append(val);
353 }
354 sops.trim(&key);
355 while (key[0] == '-') {
356 key.erase(0, 1);
357 }
358 //std::cout << "---->" << key << "<--->" << value << "<----While" << std::endl;
359 while (t + 1 < c && v[t + 1][0] != '-') {
360 value.append(" ");
361 value.append(sops.strip_qoutes(sops.trim(v[t + 1])));
362 t++;
363 }
364
365 //std::cout << "---->" << key << "<--->" << value << "<----if" << std::endl;
366 /*if (t + 1 == c - 1 && v[t + 1][0] != '-') {
367 value.append(" ");
368 value.append(sops.strip_qoutes(sops.trim(v[t + 1])));
369 t++;
370 }*/
371
372 x=t;
373 if (!options.contains(key)){
374 options.emplace(string_ops::trim(key), string_ops::trim(value));
375 options_count.emplace(key, 1);
376 //std::cout << "---->" << key << "<--->" << value << "<----does!contain" << std::endl;
377 x++;
378 continue;
379 }
380 if ( allow_multi && options.contains(key) ) {
381 options.emplace(string_ops::trim(key), string_ops::trim(value));
382 options_count[key] = options_count[key] + 1;
383 options.emplace(string_ops::trim(key), string_ops::trim(value));
384 //std::cout << "---->" << key << "<--->" << value << "<----contains" << std::endl;
385 x++;
386 continue;
387 }else{
388 throw security_exception("allow identicle keys _allow_multi is set to false duplicate key");
389 }
390
391 } else {
392 if (flag != str.npos) {
393 while (str[0] == '-') {
394 str.erase(0, 1);
395 }
396 flags.insert(str);
397 x++;
398 }
399 }
400 }
