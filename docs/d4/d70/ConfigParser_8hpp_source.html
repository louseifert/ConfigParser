304 void load_ini(string filename) {
305 if (ini_loaded) {
306 throw security_exception("you can only load an ini file once");
307 }
312 if (!std::filesystem::exists(filename) ||
313 !std::filesystem::is_regular_file(filename)) {
314 throw file_access_exception(
315 (const string) "file either does not exist or is not a regular file");
316 }
317
322 std::ofstream ofile(filename, std::ios::out);
323 if (!allow_writeable && ofile.is_open()) {
324 ofile.close();
325 throw security_exception((
326 const string) "Config files should not be writeable by the executor");
327 }
328 std::string line;
329
333 std::ifstream file(filename, std::ios::in);
334 if (!file.is_open()) {
335 throw file_access_exception(
336 "File System Error: cannot open, potentially permission");
337 }
338
339 string section_head = "";
340 while (std::getline(file, line)) {
341 if (line.empty()) {
342 continue;
343 }
344 while (line[0] == ' ' || line[0] == '\t') {
345 line.erase(0, 1);
346 }
347 if (line.size() == 0 || line[0] == '#') {
348 continue;
349 }
350 if (line[0] == ';' ||
351 (line.length() > 2 && line[0] == '/' && line[1] == '/')) {
352 continue;
353 }
354 /*TODO add regex check here*/
355 if (line[0] == '[') {
356 section_head = line.substr(1, line.find_first_of(']') - 1) + ".";
357 continue;
358 }
359
360 const size_t s = line.find_first_of("=", 0);
361 if (s == line.npos) {
362 flags.insert(line);
363 continue;
364 }
365
366 string key = line.substr(0, s);
367 string value = line.substr(s + 1, line.length());
368 string_ops::strip_qoutes(&key);
369 string_ops::strip_qoutes(&value);
370 key = section_head + key;
371
372 if (!stringops.strip_check(&key) || !stringops.strip_check(&value)) {
373 restricted.emplace(key, value);
374 restricted_count++;
375 } else {
376 if (options.contains(key)) {
377 options[key] = value;
378 } else {
379 options.emplace(key, value);
380 ini_parm_count++;
381 }
382 }
383 }
384 ini_loaded = true;
385 }

422 void parse(int c, char *v[]) {
423 string prev = "";
424 for (int i = 0; i < c; i++) {
425 string str(v[i]);
426 string key;
427 string value;
428
429 size_t flag = str.find_first_of('-', 0);
430 size_t assigner = str.find_first_of('=', 0);
431 // if (flag == assigner) {
432 // continue;
433 // }
434
435 if (flag == 0 && assigner >= str.npos) {
436 while (str[0] == '-') {
437 str.erase(0, 1);
438 }
439 prev = str;
440 flags.insert(str);
441 continue;
442 } else if (flag == 0 && assigner != string::npos) {
443 // Case where input is a -something=something
444 key = str.substr(1, assigner - 1);
445 value = str.substr(assigner + 1, str.length() - assigner - 1);
446 if (!options.contains(key)) {
447 options.emplace(key, value);
448 }
449 continue;
450 }
451 }
