161 void load_ini(string filename) {
162 if (ini_loaded) {
163 throw security_exception("you can only load an ini file once");
164 }
169 if (!std::filesystem::exists(filename) ||
170 !std::filesystem::is_regular_file(filename)) {
171 throw file_access_exception(
172 (const string) "file either does not exist or is not a regular file");
173 }
174
179 std::ofstream ofile(filename, std::ios::out);
180 if (!allow_writeable && ofile.is_open()) {
181 ofile.close();
182 throw security_exception((
183 const string) "Config files should not be writeable by the executor");
184 }
185 std::string line;
186
190 std::ifstream file(filename, std::ios::in);
191 /*not sure this condition could happen?
192 * if (!file.is_open()) {
193 throw file_access_exception(
194 "File System Error: cannot open, potentially permission");
195 }*/
196
197 string section_head = "";
198 while (std::getline(file, line)) {
199 if (line.empty()) {
200 continue;
201 }
202 while (line[0] == ' ' || line[0] == '\t') {
203 line.erase(0, 1);
204 }
205 if (line.size() == 0 || line[0] == '#') {
206 continue;
207 }
208 if (line[0] == ';' ||
209 (line.length() > 2 && line[0] == '/' && line[1] == '/')) {
210 continue;
211 }
212 /*TODO add regex check here*/
213 if (line[0] == '[') {
214 section_head = line.substr(1, line.find_first_of(']') - 1) + ".";
215 continue;
216 }
217
218 const size_t s = line.find_first_of("=", 0);
219 if (s == line.npos) {
220 flags.insert(line);
221 continue;
222 }
223
224 string key = line.substr(0, s);
225 string value = line.substr(s + 1, line.length());
226 string_ops::strip_qoutes(&key);
227 string_ops::strip_qoutes(&value);
228 key = section_head + key;
229
230 if (!stringops.strip_check(&key) || !stringops.strip_check(&value)) {
231 restricted.emplace(key, value);
232 restricted_count++;
233 } else {
234 if (options.contains(key)) {
235 options[key] = value;
236 } else {
237 options.emplace(key, value);
238 ini_parm_count++;
239 }
240 }
241 }
242 ini_loaded = true;
243 }

282 void parse(int c, char *v[]) {
283 string prev = "";
284 for (int i = 0; i < c; i++) {
285 string str(v[i]);
286 string key;
287 string value;
288
289 size_t flag = str.find_first_of('-', 0);
290 size_t assigner = str.find_first_of('=', 0);
291 // if (flag == assigner) {
292 // continue;
293 // }
294
295 if (flag == 0 && assigner >= str.npos) {
296 while (str[0] == '-') {
297 str.erase(0, 1);
298 }
299 if ((i + 1 < c && v[i + 1][0] == '-') || i + 1 >= c) {
300 flags.insert(str);
301 continue;
302 } else if (i + 1 < c) {
303 std::string key(str);
304 std::string value(v[i + 1]);
305 int t = i;
306 while (t + 1 < c && v[t + 1][0] != '-') {
307 value.append(" ");
308 value.append(v[t]);
309 t++;
310 }
311 i = t;
312 options.emplace(key, value);
313 }
314 } else if (flag == 0 && assigner != string::npos) {
315 // Case where input is a -something=something
316 key = str.substr(1, assigner - 1);
317 value = str.substr(assigner + 1, str.length() - assigner - 1);
318 if (!options.contains(key)) {
319 options.emplace(key, value);
320 }
321 continue;
322 }
323 }
324 }
